SQL> CREATE OR REPLACE TRIGGER before_delete_orders
  2  BEFORE DELETE ON orders
  3  FOR EACH ROW
  4  BEGIN
  5      IF :OLD.ORDER_STATUS = 'PENDING' THEN
  6          RAISE_APPLICATION_ERROR(-20002, 'Data Integrity Error: Cannot delete an order while it is Pending.');
  7      END IF;
  8  END;

7. // COMPOUND TRIGGER on Service Record


SQL> CREATE OR REPLACE TRIGGER compound_trigger_service_record
  2  FOR INSERT OR DELETE ON service_record
  3  COMPOUND TRIGGER
  4
  5      BEFORE EACH ROW IS
  6      BEGIN
  7          IF INSERTING THEN
  8              DBMS_OUTPUT.PUT_LINE('Preparing to insert Service Record: ' || :NEW.SERVICE_RECORD_ID);
  9          END IF;
 10      END BEFORE EACH ROW;
 11
 12      AFTER EACH ROW IS
 13      BEGIN
 14          IF DELETING THEN
 15              DBMS_OUTPUT.PUT_LINE('Service Record ' || :OLD.SERVICE_RECORD_ID || ' has been deleted.');
 16          END IF;
 17      END AFTER EACH ROW;
 18
 19  END compound_trigger_service_record;
 20  /

8. // STATEMENT-LEVEL TRIGGER on Orders


SQL> CREATE OR REPLACE TRIGGER statement_trigger_orders
  2  BEFORE INSERT OR DELETE ON orders
  3  DECLARE
  4      v_msg VARCHAR2(100);
  5  BEGIN
  6      IF INSERTING THEN
  7          v_msg := 'A new order is being inserted into Orders table.';
  8      END IF;
  9
 10      IF DELETING THEN
 11          v_msg := 'An order is being deleted from Orders table.';
 12      END IF;
 13
 14      DBMS_OUTPUT.PUT_LINE(v_msg);
 15  END statement_trigger_orders;

9. // Archive on Delete (Insert into Deleted_Orders when order is deleted)


SQL> CREATE OR REPLACE TRIGGER trg_archive_orders
  2  BEFORE DELETE ON orders
  3  FOR EACH ROW
  4  BEGIN
  5      INSERT INTO Deleted_Orders (
  6          ORDER_ID, CUSTOMER_ID, ORDER_STATUS,
  7          TOTAL_AMOUNT, DISCOUNT, EMPLOYEE_ID, ORDER_DATE
  8      )
  9      VALUES (
 10          :OLD.ORDER_ID, :OLD.CUSTOMER_ID, :OLD.ORDER_STATUS,
 11          :OLD.TOTAL_AMOUNT, :OLD.DISCOUNT, :OLD.EMPLOYEE_ID, :OLD.ORDER_DATE
 12      );
 13  END;
 14  /
