-- CASE WHEN VIEW (CUSTOMER CITY CATEGORY)

SQL> CREATE VIEW customer_city_type AS
  2  SELECT CUSTOMER_ID,
  3         NAME,
  4         ADDRESS_CITY,
  5         CASE
  6             WHEN ADDRESS_CITY = 'Chennai' THEN 'METRO CITY'
  7             WHEN ADDRESS_CITY = 'Madurai' THEN 'TEMPLE CITY'
  8             ELSE 'OTHER CITY'
  9         END AS CITY_TYPE
 10  FROM customer;

View created.

-- CASE WHEN VIEW (EMPLOYEE SALARY CATEGORY)

SQL> CREATE VIEW employee_salary_grade AS
  2  SELECT EMPLOYEE_ID,
  3         NAME,
  4         SALARY,
  5         CASE
  6             WHEN SALARY >= 30000 THEN 'HIGH SALARY'
  7             WHEN SALARY >= 25000 THEN 'MEDIUM SALARY'
  8             ELSE 'LOW SALARY'
  9         END AS SALARY_GRADE
 10  FROM employee;

View created.

SQL> SELECT * FROM employee_salary_grade;

----------- -------- ------ -------------
101         RAMESH   30000  HIGH SALARY
102         SURESH   25000  MEDIUM SALARY
